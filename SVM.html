#import statements
import pandas as pd
import numpy as np
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
from sklearn.svm import SVC
import from sklearn.metrics import accuracy_score, recall_score, precision_score, confusion_matrix, classification_report


#load parkinsons dataset and understand its dimensions
df = pd.read_csv('sample_data/parkinsons.csv')
print('Shape :',df.shape)
df.head()

#Shape : (195, 24)

#check for missing values from dataset
df.isnull().sum()

#No missing or null value in dataset

# Separate features and target variable
features = df.drop('status', axis=1)
features = features.drop('name',axis=1)
target = df['status']

# Get the distribution of the target variable
target_distribution = target.value_counts(normalize=True) * 100
target_distribution

# Split data into training and testing sets
X_train, X_test, y_train, y_test = train_test_split(features, target, test_size=0.2, random_state=42)

#standardization on dataset
scaler = StandardScaler()
X_train = scaler.fit_transform(X_train)
X_test = scaler.transform(X_test)

#svm model training
svm_model = SVC(kernel='linear')
svm_model.fit(X_train, y_train)

#make pridiction
y_pred = svm_model.predict(X_test)
print(y_pred)

#evaluating the model
accuracy = accuracy_score(y_test, y_pred)
recall = recall_score(y_test, y_pred)
precision = precision_score(y_test, y_pred)
confusion = confusion_matrix(y_test, y_pred)

#report on evaluation
print("Accuracy:", accuracy)
print("Recall:", recall)
print("Precision:", precision)
print("Confusion Matrix:\n", confusion)
print(classification_report(y_test,y_pred))
true_positive_rate = confusion[1, 1] / (confusion[1, 1] + confusion[1, 0])
print("True Positive Rate:", true_positive_rate)
